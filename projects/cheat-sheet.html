#include?(value) Returns true if any element == value
#sort sort{|a, b|...}
Returns array sorted by <=> operator or by block.

a = ["d", "a", "e", "c", "b"]
a.sort => ["a", "b", "c", "d", "e"]
a.sort{|x,y| y <=> x } => ["e", "d", "c", "b", "a"]
#flatten!(level)
#flatten 		Destructive method that pushes nested arrays into one array.

a = [1,2,[3,4],5,[6,[7,8]]]
a.flatten! = [1,2,3,4,5,6,7,8]
a.flatten!(1) = [1,2,3,4,5,[6,[7,8]]]
#map!{|item| block}
#map 		Invokes the given block once for each element of self,
replacing the element with the value returned by the block.

a = [ "a", "b", "c"]
a.map! {|x| x + "!" }
a = ["a!", "b!", "c!"]
#reverse
#reverse! 		Reverse given array destructively or non-destructively.

a = ['a', 'b', 'c']
a.reverse => ['c', 'b', 'a']
a = ['a', 'b' 'c']

a = ['a', 'b', 'c']
a.reverse! => ['c', 'b', 'a']
a = ['c', 'b', 'a']
#select{|item| block} Returns a new array containing all elements of array
for which the given block returns a true value.

[1,2,3,4,5].select {|num| num.even?} => [2,4]
